{{/* Renders chord markers inside song text. @param {string} mode The rendering mode. Supported values: "inline" and "lyrics". @param {string} [modifier] A CSS class modifier appended as "chords--". @reference https://developer.mozilla.org/en-US/docs/Web/HTML/Reference/Elements/ruby @example {{% chords inline %}} **[Intro]**: `[Am]` `[F]` `[C]` `[E]` {{% /chords %}} @example {{% chords lyrics %}} `[Am]`Hello, wor`[F]`ld {{% /chords %}} Marker syntax: wo`[Am]`rd → chord inside a word; "wo" stays outside ruby, "rd" anchors the chord `[Am]`word → chord before the following word/syllable word`[Am]` → chord after the preceding word `[Am]` → standalone chord Rendering: RE2 has no conditional replacement, so rendering is two-phase. Phase 1 — tokenize all chord markers into an intermediate format: prefix`[chord]`suffix → CHORD_BEGIN;prefix;chord;suffix;CHORD_END Phase 2 — render tokens according to the selected mode. Inline mode — one replacement pass: prefix + chord + suffix → prefixchordsuffix Lyrics mode — four replacement passes, in code order: prefix + chord + suffix → prefixsuffixchord chord only → chord prefix + chord → prefixchord chord + suffix → suffixchord */}} {{- /* Get arguments. */}} {{- $mode := cond .IsNamedParams (.Get "mode") (.Get 0) }} {{- $modifier := cond .IsNamedParams (.Get "modifier") (.Get 1) }} {{- if not $mode }}{{ errorf "The %q shortcode requires a mode argument, 'inline' or 'lyrics': see %s" $.Name $.Position }}{{ end }} {{- /* Build output. */}} {{- $out := .Inner | strings.ReplaceRE "([^\\s\\x60]*)\\x60\\[(.+?)\\]\\x60([^\\s\\x60]*)" "CHORD_BEGIN;$1;$2;$3;CHORD_END" }} {{- if eq $mode "inline" }} {{- $out = $out | strings.ReplaceRE "CHORD_BEGIN;([^;]*);([^;]+);([^;]*);CHORD_END" "$1$2$3" }} {{- else if eq $mode "lyrics" }} {{- $out = $out | strings.ReplaceRE "CHORD_BEGIN;([^;]+);([^;]+);([^;]+);CHORD_END" "$1$3$2" }} {{- $out = $out | strings.ReplaceRE "CHORD_BEGIN;;([^;]+);;CHORD_END" "$1" }} {{- $out = $out | strings.ReplaceRE "CHORD_BEGIN;([^;]+);([^;]+);;CHORD_END" "$1$2" }} {{- $out = $out | strings.ReplaceRE "CHORD_BEGIN;;([^;]+);([^;]+);CHORD_END" "$2$1" }} {{- end }} {{- /* Render. */}}
